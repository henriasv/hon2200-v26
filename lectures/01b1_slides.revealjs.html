import pandas as pd
df = pd.DataFrame({
    "talent" : [4, 5, 3, 4, 6, 2],
    "innsats" : [2, 4, 3, 7, 4, 1],
    "resultat" : [24, 73, 25, 204, 93, 4]
})
print(df)
